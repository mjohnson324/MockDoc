Doctor Heroku ids: 698 to 807

docs = Doctor.all
doc_ids = []
docs.each { |doc| doc_ids << doc.id }

app_addresses = {}

doc_ids.each do |id|
  app = Appointment.where(doctor_id: id).limit(1)
  app_addresses[id] = app[0].address
end

doc_ids.each { |id| Doctor.find(id).update(address: app_addresses[id]) }
